<script>

// DOM Elements

const stage1 = document.getElementById('stage1');

const stage2 = document.getElementById('stage2');

const stage3 = document.getElementById('stage3');

const typingText = document.getElementById('typingText');

const interactionArea = document.getElementById('interactionArea');

const noBtn = document.getElementById('noBtn');

const progressBar = document.getElementById('progressBar');

// Configuration

const message = "Hey beautiful... I've been meaning to ask you something very important. Will you be my Valentine?";

const typingSpeed = 50;

// Transition to Letter

function openEnvelope() {

// Animate out stage 1

stage1.style.opacity = '0';

stage1.style.transform = 'scale(0.8)';

setTimeout(() => {

stage1.style.display = 'none';

stage2.classList.add('active');

// Start typing effect after small delay

setTimeout(() => {

progressBar.style.width = '100%';

typeWriter();

}, 500);

}, 500);

}

// Typewriter Effect

let charIndex = 0;

function typeWriter() {

if (charIndex < message.length) {

typingText.innerHTML += message.charAt(charIndex);

charIndex++;

setTimeout(typeWriter, typingSpeed);

} else {

// Typing finished, show buttons

document.querySelector('.cursor').style.display = 'none';

interactionArea.classList.remove('opacity-0');

}

}

// The "Impossible No" Logic

function moveButton(e) {

const container = document.getElementById('interactionArea');

const containerRect = container.getBoundingClientRect();

const btnRect = noBtn.getBoundingClientRect();

// Calculate new position bounded by container/screen

// We'll use fixed positioning relative to viewport to ensure it can jump anywhere safely or absolute relative to parent

// Simple random logic

const maxX = window.innerWidth - btnRect.width - 40;

const maxY = window.innerHeight - btnRect.height - 40;

let newX = Math.random() * maxX;

let newY = Math.random() * maxY;

// Ensure it doesn't go off screen (clamping)

newX = Math.max(20, Math.min(newX, maxX));

newY = Math.max(20, Math.min(newY, maxY));

// Apply fixed position to break out of layout flow

noBtn.style.position = 'fixed';

noBtn.style.left = newX + 'px';

noBtn.style.top = newY + 'px';

// Add frustrated emojis to button text occasionally

const texts = ["No ðŸ™„", "Are you sure?", "Really?", "Try again!", "Too slow!", "Nope!"];

noBtn.innerText = texts[Math.floor(Math.random() * texts.length)];

}

// Event listeners for desktop and mobile

noBtn.addEventListener('mouseover', moveButton);

noBtn.addEventListener('touchstart', (e) => {

e.preventDefault(); // prevent click

moveButton();

});

noBtn.addEventListener('click', (e) => {

e.preventDefault();

moveButton();

});

// Success & Celebration

function acceptLove() {

// Trigger Confetti

fireConfetti();

// Start Heart Rain

createHeartRain();

// Transition UI

stage2.style.opacity = '0';

setTimeout(() => {

stage2.style.display = 'none';

stage3.classList.remove('hidden');

stage3.style.display = 'flex';

// More confetti on success screen load

setTimeout(fireConfetti, 300);

}, 500);

}

// Confetti Function

function fireConfetti() {

const duration = 3000;

const end = Date.now() + duration;

(function frame() {

confetti({

particleCount: 5,

angle: 60,

spread: 55,

origin: { x: 0 },

colors: ['#fb7185', '#be123c', '#ffd700']

});

confetti({

particleCount: 5,

angle: 120,

spread: 55,

origin: { x: 1 },

colors: ['#fb7185', '#be123c', '#ffd700']

});

if (Date.now() < end) {

requestAnimationFrame(frame);

}

}());

}

// Heart Rain Animation

function createHeartRain() {

const container = document.body;

const hearts = ['ðŸ’–', 'ðŸ¥°', 'ðŸŒ¹', 'ðŸ’‘', 'ðŸ’Œ'];

for (let i = 0; i < 30; i++) {

setTimeout(() => {

const heart = document.createElement('div');

heart.classList.add('heart-rain');

heart.innerHTML = hearts[Math.floor(Math.random() * hearts.length)];

heart.style.left = Math.random() * 100 + 'vw';

heart.style.animationDuration = (Math.random() * 2 + 3) + 's';

heart.style.fontSize = (Math.random() * 1 + 1) + 'rem';

container.appendChild(heart);

// Cleanup

setTimeout(() => {

heart.remove();

}, 5000);

}, i * 100);

}

}
